\begin{defn}[Fibonacci Sequence]
	The sequence is defined by 
	\[ 
		F_0=0, F_1=1, F_{n+1} = F_n + F_{n-1} 
	\]
	
\end{defn}
\begin{thm}
\[ 
\lim_{n \to  + \infty} \frac{F_{n+1} }{F_n} = \phi
\]

\end{thm}

\[ 
	F( x) = \frac{x}{1-x -x^{2}}

 \]

 		


\end{proof}
